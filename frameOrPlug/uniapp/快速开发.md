#### 复制

```js
exports.copys = function(word, cb) {
	try{
		var os = plus.os.name;

		if('iOS' == os) {
			var pasteboard = plus.ios.invoke('UIPasteboard', 'generalPasteboard');
			plus.ios.invoke(pasteboard, 'setValue:forPasteboardType:', word, 'public.utf8-plain-text');
		} else {
			var main = plus.android.runtimeMainActivity();
			var clip = main.getSystemService('clipboard');
			plus.android.invoke(clip, 'setText', word);
		}

		this.showToast($lang.copySuccessfully);
		cb && cb()
	} catch (e) {
		this.showToast($lang.copyFailed);
	}
}
```
