实时更新数据 - sync 

子组件

```
<li @click.stop="setOption(item)" :key="index">{{ item.text }}</li>

props: {
    id: {
        type: [Number, String]
    }
},

setOption (item) {
    this.selectedItem = item
    this.status = false
    this.$emit('update:id', item.id)
},
```

父组件

```
<my-select class="fr" :id.sync="id"></my-select>
```
